{ “status”: 0, “statusinfo”: “正常”, “defaultquality”: 10 “qualityDescription”: [@“清晰”] “qualities”: [{ “quality”: 10, “desp”: “清晰”, “playurl”: “http://bm2.33.play.bokecc.com/flvs/ca/Qxc52/uolaj09ZUD-10.mp4?t=1396457244&key=7A9DEB2C7BEFC4EFC4046CC666040054&upid=4109151396450044835” }] }
